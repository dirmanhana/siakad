function automatic(nbr, max, delay) {
   if (nbr <= max) {
      document.frames["myInnerFrame"].location.href = nbr + ".html";
   }
   if (nbr < max) {
      setTimeout("automatic("+(nbr++)","+max+","+delay+")", delay); 
   }
}
